case 1) 이분탐색으로 key 삽입 - "시간초과"

``` swift
import Foundation

let T = Int(readLine()!)!

for _ in 0..<T {
    let k = Int(readLine()!)!
    var Q = [Int]()
    
    for _ in 0..<k {
        let input = readLine()!.split(separator: " ").map { String($0) }
        
        switch input[0] {
            case "I":
            insertQ(Int(input[1])!, &Q)
            default:
            deleteQ(Int(input[1])!, &Q)
        }
    }
    
    guard !Q.isEmpty else {
        print("EMPTY")
        continue
    }
    
    print("\(Q.last!) \(Q.first!)")
}

// 이분탐색으로 정렬해서 key 삽입
func insertQ(_ key: Int, _ q: inout [Int]) {
    guard !q.isEmpty else { 
        q.append(key) 
        return
    }
    
    var left = 0
    var right = q.count - 1
    
    while left < right {
        let mid = (left + right) / 2
        
        if q[mid] == key {
            q.insert(key, at: mid)
            return
        }
        else if q[mid] < key {
            left = mid + 1
        }
        else {
            right = mid - 1
        }
    }
    
    if q[left] > key {
        q.insert(key, at: left)
    }
    else {
        q.insert(key, at: left + 1)
    }
}

func deleteQ(_ key: Int, _ q: inout [Int]) {  // 1: 최대, -1: 최소
    guard !q.isEmpty else { return }
    
    if key == -1 {
        q.removeFirst()
    }
    else {
        q.removeLast()
    }
}
```
