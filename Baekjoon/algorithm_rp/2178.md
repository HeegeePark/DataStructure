# 2178) 미로 탐색

``` python
from collections import deque
import sys
input = sys.stdin.readline

dy = [0, 0, 1, -1]
dx = [1, -1, 0, 0]

def is_valid(y, x):
    return 0 <= y < N and 0 <= x < M

N, M = map(int, input().split())
board = [input() for _ in range(N)]
checkArray = [[False] * M for _ in range(N)]

q = deque()
q.append((0, 0, 1)) # 도착은 n-1, m-1
checkArray[0][0] = True

while q:
    y, x, d = q.popleft()
    
    if y == N - 1 and x == M - 1:
        print(d)
        break
        
    for i in range(4):
        ny = y + dy[i]
        nx = x + dx[i]
        nd = d + 1
        
        if not is_valid(ny, nx):
            continue
        
        if checkArray[ny][nx]:
            continue
        
        if board[ny][nx] == '0':
            continue
        
        checkArray[ny][nx] = True
        q.append((ny, nx, nd))
```
