import Foundation

let dx = [0, 0, 1, -1]
let dy = [1, -1, 0, 0]
var (M, N, K) = (0, 0, 0)
var map = [[Int]]()
var visited = [[Bool]]()

let T = Int(readLine()!)!

for _ in 0..<T {
    let mnk = readLine()!.split(separator: " ").map { Int($0)! }
    (M, N, K) = (mnk[0], mnk[1], mnk[2])
    map = Array(repeating: [Int](repeating: 0, count: M), count: N)
    visited = Array(repeating: [Bool](repeating: false, count: M), count: N)
    var cabbages = [(Int, Int)]()
    var answer = 0
    
    for _ in 0..<K {
        let cab = readLine()!.split(separator: " ").map { Int($0)! }
        cabbages += [(cab[1], cab[0])]
        map[cab[1]][cab[0]] = 1
    }
    
    for (r, c) in cabbages {
        if !visited[r][c] {
            bfs(r, c)
            answer += 1
        }
    }
    print(answer)
}

func bfs(_ r: Int, _ c: Int) {
    var q = [(r: Int, c: Int)]()
    q.append((r, c))
    visited[r][c] = true
    
    while !q.isEmpty {
        let curr = q.removeFirst()
        
        for d in zip(dx, dy) {
            let nr = curr.r + d.0
            let nc = curr.c + d.1
            
            guard isValid(nr, nc) && !visited[nr][nc] && map[nr][nc] == 1 else {
                continue
            }
            
            visited[nr][nc] = true
            q.append((nr, nc))
        }
    }
}

func isValid(_ r: Int, _ c: Int) -> Bool {
    return 0..<N ~= r && 0..<M ~= c
}
