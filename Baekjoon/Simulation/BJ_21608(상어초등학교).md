``` swift
// 상어초등학교
import Foundation

let xDir = [0, 0, -1, 1]
let yDir = [-1, 1, 0, 0]

let N = Int(readLine()!)!
var answer = 0

var seats = [[Int]](repeating: [Int](repeating: 0, count: N), count: N)
var like = [[Int]](repeating: [Int](repeating: 0, count: 4), count: N * N + 1)
var stud = [Int](repeating: 0, count: N * N + 1)

for i in 0..<N * N {
    let input = readLine()!.split(separator: " ").map { Int($0)! }
    
    let student = input[0]
    stud[i] = student
    
    for j in 0..<4 {
        like[student][j] = input[j + 1]
    }
}

// batch
for s in 0..<N * N {
    let student = stud[s]
    var sInj = -1, sEmp = -1, sy = -1, sx = -1

    for i in 0..<N {
        for j in 0..<N {
            if seats[i][j] != 0 {
                continue
            }

            var inj = 0, emp = 0

            for dir in 0..<4 {
                let dy = i + yDir[dir]
                let dx = j + xDir[dir]

                guard 0..<N ~= dy && 0..<N ~= dx else {
                    continue
                }

                if seats[dy][dx] == 0 {
                    emp += 1
                    continue
                }

                for k in 0..<4 {
                    if seats[dy][dx] == like[student][k] {
                        inj += 1
                        break
                    }
                }
            }

            // conditions
            if inj > sInj {
                sInj = inj
                sEmp = emp
                sy = i
                sx = j
            } else if inj == sInj {
                if emp > sEmp {
                    sy = i
                    sx = j
                    sInj = inj
                    sEmp = emp
                }
            } else {
                continue
            }
        }
    }

    seats[sy][sx] = student
}

// satisfy
for i in 0..<N {
    for j in 0..<N {
        let student = seats[i][j]
        var ret = 0

        for dir in 0..<4 {
            let dy = i + yDir[dir]
            let dx = j + xDir[dir]

            guard 0..<N ~= dy && 0..<N ~= dx else {
                continue
            }

            for k in 0..<4 {
                if seats[dy][dx] == like[student][k] {
                    ret += 1
                    break
                }
            }
        }

        if ret > 0 {
            var ans = 1

            for _ in 0..<ret - 1 {
                ans *= 10
            }

            answer += ans
        }
    }
}

print(answer)
```
