``` swift
import Foundation

let N = Int(readLine()!)!

var dequeue = [Int]()
var enqueue = Array(1...N)
var flag = true

@discardableResult
func popFirst() -> Int {
    if dequeue.isEmpty {
        dequeue = enqueue.reversed()
        enqueue.removeAll()
    }
    return dequeue.removeLast()
}

func last() -> Int {
    var last: Int?
    if enqueue.isEmpty {
        dequeue.reverse()
        last = dequeue.popLast()
        dequeue.reverse()
    }
    else {
        last = enqueue.popLast()
    }
    return last!
}

while dequeue.count + enqueue.count > 1 {
    if flag {   // 버리기
        popFirst()
    }
    else {
        let num = popFirst()
        enqueue.append(num)
    }
    flag.toggle()
}

print(last())
```
