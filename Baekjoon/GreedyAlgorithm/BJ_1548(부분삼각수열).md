``` swift
import Foundation

let N = Int(readLine()!)!
let A = readLine()!.split(separator: " ").map { Int($0)! }.sorted(by: <)
var answer = [Int]()

if N < 3 {
    print(N)
    exit(0)
}

for i in 0..<N-1 {
    for j in i+2..<N {
        if A[i] + A[i+1] > A[j] {
            answer.append(j-i+1)
        }
    }
}

print(answer.isEmpty ? 2: answer.max()!)
```
